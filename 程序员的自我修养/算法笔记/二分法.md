# 二分法

满足条件写：

1. l = mid 或 r = mid

2. mid首先写成(l + r) / 2。

3. 如果l = mid，那么mid条件+1，即mid = (l+ r + 1) / 2 = 1。

4. l = mid对应r = mid - 1

5. r = mid对应l = mid + 1

<br/>

```java
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int n = nums.size();

        vector<int> ans(2, -1);
        if (n == 0) return ans;
        
        int l = 0, r = n - 1;
        while (l < r) {
            int mid = l + r >> 1;
            if (nums[mid] >= target) r = mid;
            else l = mid + 1;
        }
        if (nums[r] != target) return ans;
        ans[0] = r;
        l = 0, r = n - 1;
        while (l < r) {
            int mid = l + r + 1 >> 1;
            if (nums[mid] <= target) l = mid;
            else r = mid - 1;
        }
        ans[1] = r;

        return ans;
    }
};
```
